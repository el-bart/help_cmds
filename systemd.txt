# systemd[1]: start request repeated too quickly for foobar.service
systemctl reset-failed foobar.service
systemctl restart      foobar.service

# list user units, etc.
systemctl --user
# list user services
systemctl --user list-unit-files --type=service
# stop user service
systemctl --user stop foo-bar

# restart OS w/o rebooting PC
systemctl soft-reboot

# overriding foobar service
systemctl edit foobar
# will edit:
/etc/systemd/system/foobar.service.d/override.conf
