PermitEmptyPasswords no
KerberosOrLocalPasswd no
EOF

# I2C usage
(enable I2C in `raspi-config` first!)

## scan bus for devices
i2cdetect -y 1

## read entire from EEPROM
i2cdump -y 1 0x50

## read byte 0x42 from EEPROM (0x50 address)
i2cget -y 1 0x50 0x42

## set byte 0x42 to 0x12 (EEPRIN is at 0x50 address)
i2cset -y 1 0x50 0x42 0x12

#
# RasPi CSI camera
#

# install all deps (e.g. libav)
apt install rpicam-apps

# list cameras
rpicam-vid  --list-cameras

# record video (1k)
rpicam-vid --mode 1920:1080:8:U -o x.mkv

# capture image into a file
rpicam-jpeg -o x.jpg

# streaming camera video
rpicam-vid -t 0 -n --inline --listen  -o tcp://0.0.0.0:1234

# receiving stream
ffplay tcp://my-camera-address:1234 -fflags nobuffer -flags low_delay -framedrop

# streaming using MPEG-TS
rpicam-vid -t 0 -n --codec libav --libav-format mpegts -o 'tcp://0.0.0.0:1234?listen=1'

# recodinf MPEG-TS stream
ffmpeg -i tcp://my-camera-address:1234 -c copy output.ts


#
# RasPi USB as eth card
#
https://www.raspberrypi.com/news/usb-gadget-mode-in-raspberry-pi-os-ssh-over-usb/
https://github.com/raspberrypi/rpi-usb-gadget

#
# before 1st run
#

# 1. mount /boot partition

# 2. edit / create "user-data" (cloud-init format) and append:
```
rpi:
  enable_usb_gadget: true
enable_ssh: true  # Optional but recommended when using gadget mode
```

# 3. cfg also user and keys:
```
users:
  - name: pi
    gecos: Raspberry Pi User
    sudo: ALL=(ALL) NOPASSWD:ALL
    shell: /bin/bash
    ssh_authorized_keys:
      - ssh-ed25519 AAAA... user@example.com
    lock_passwd: true
    passwd: $6$rounds=4096$...   # optional hashed password
ssh_pwauth: false
disable_root: true
```
